ProcessingでWebカメラが使えないとき
GifAnimationの作り方 1) 準備
Gifアニメーションを作るためには新しくライブラリをインポートする必要があります。Processingを開いて、sketchタブ>Import Library>Add LibraryでGifAnimationを検索、Installする。
2) サンプルコード 
						
import gifAnimation.*;
GifMaker gifMaker;
void setup() {
  size(500, 500, P3D);
  gifMaker = new GifMaker(this, "box.gif");
  gifMaker.setRepeat(0);
  gifMaker.setDelay(30);
}
int time = 0;
void draw() {
  background(255);
  translate(width/2, height/2, 0);
  rotate(radians(3*time), sin(radians(time)), cos(radians(time)), 1);
  fill(0, 0, 255);
  box(160);
  ++time;
  gifMaker.addFrame();
  if (radians(time) >= TWO_PI) {
	gifMaker.finish();
	exit();
  }
}
